CREATE TABLE Dessert(
Dessert_Id INT,
Name VARCHAR(255), 
Ingredients LONGTEXT,  
Vegan BOOLEAN, 
Vegetarian BOOLEAN, 
Keto BOOLEAN,
Calories INT,
Crabs INT, 
TotalFat INT,
Cholesterol INT, 
Sodium INT, 
Sugar INT, 
Protein INT,
User_Id INT NOT NULL, 
CONSTRAINT Dessert_PK PRIMARY KEY (Dessert_Id)
);
