Create Table User(
User_Id INT NOT NULL, 
First_Name VARCHAR(255) NOT NULL,
Last_Name VARCHAR(255) NOT NULL, 
Age INT, 
Weight INT,
Height INT, 
Vegan BOOLEAN, 
Vegetarian BOOLEAN, 
Keto BOOLEAN,
Breakfast_Id INT, 
Lunch_Id INT, 
Dinner_Id INT, 
Dessert_Id INT, 
Snack_Id INT, 
CONSTRAINT User_PK PRIMARY KEY (User_Id)
);
